<script>
    const STORAGE_KEY = 'recentTicketIDs';
    const MAX_HISTORY = 10;
    const BASE_URL = "https://support.sevensuite.com.br/TicketDossier/";
    const SUFFIX = "/OpenBySequentialId";

    function getRecentIDs() {
        const data = localStorage.getItem(STORAGE_KEY);
        return data ? JSON.parse(data) : [];
    }

    function updateHistory(newIds) {
        let history = getRecentIDs();
        for (const id of newIds.reverse()) {
            history = history.filter(item => item !== id);
            history.unshift(id);
        }
        if (history.length > MAX_HISTORY) {
            history = history.slice(0, MAX_HISTORY);
        }
        localStorage.setItem(STORAGE_KEY, JSON.stringify(history));
        renderRecentList();
    }

    function renderRecentList() {
        const listElement = document.getElementById('recentList');
        const history = getRecentIDs();
        listElement.innerHTML = '';

        if (history.length === 0) {
            listElement.innerHTML = '<li>Nenhum chamado recente.</li>';
            return;
        }

        history.forEach(id => {
            const fullURL = BASE_URL + id + SUFFIX;
            const listItem = document.createElement('li');
            listItem.innerHTML = `<a href="${fullURL}" target="_blank">${id}</a>`;
            listElement.appendChild(listItem);
        });
    }

    function openTickets() {
        const input = document.getElementById('idInput').value;
        let ids = input.split(/[\s,]+|\n/)
                       .map(id => id.trim())
                       .filter(id => id && !isNaN(id));

        if (ids.length === 0) {
            alert('Por favor, digite pelo menos um ID válido.');
            return;
        }

        const uniqueIds = [...new Set(ids)];

        uniqueIds.forEach(id => {
            const fullURL = BASE_URL + id + SUFFIX;
            window.open(fullURL, '_blank');
        });

        updateHistory(uniqueIds);
        document.getElementById('idInput').value = '';
    }

    function updateCountdown() {
        const now = new Date();
        const today = now.getDate();
        const tomorrow = today + 1;

        let targetTime;
        let targetLabel;

        const target17h = new Date(now.getFullYear(), now.getMonth(), today, 17, 0, 0);
        const target18h = new Date(now.getFullYear(), now.getMonth(), today, 18, 0, 0);
        const nextDay17h = new Date(now.getFullYear(), now.getMonth(), tomorrow, 17, 0, 0);

        if (now < target17h) {
            targetTime = target17h;
            targetLabel = "para as 17:00";
        } else if (now < target18h) {
            targetTime = target18h;
            targetLabel = "para as 18:00 (Fim do Expediente)";
        } else {
            targetTime = nextDay17h;
            targetLabel = "para 17:00 de amanhã";
        }

        const diff = targetTime - now;
        const countdownElement = document.getElementById('countdown');
        if (!countdownElement) return;

        if (diff <= 0) {
            countdownElement.innerHTML = `<span style="color: #FF9800;">Objetivo atingido!</span>`;
        } else {
            const hours = Math.floor(diff / (1000 * 60 * 60));
            const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
            const seconds = Math.floor((diff % (1000 * 60)) / 1000);

            const timeString =
                `${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;

            countdownElement.innerHTML = `${targetLabel}: <span style="color: #FF9800;">${timeString}</span>`;
        }
    }

    document.addEventListener('DOMContentLoaded', () => {
        renderRecentList();
        updateCountdown();
        setInterval(updateCountdown, 1000);
    });
</script>
